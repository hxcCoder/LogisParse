&quot;&quot;&quot;FastAPI dependency injections.

All dependencies are defined here and explicitly passed to endpoints via Depends().
This ensures:
- No global state in imports
- Full testability via dependency_overrides
- Explicit dependency graph
- Type safety
&quot;&quot;&quot;

import logging
from collections.abc import AsyncGenerator
from typing import Annotated

from fastapi import Depends, Request
from fastapi.security import HTTPAuthorizationCredentials, HTTPBearer
from sqlalchemy.ext.asyncio import AsyncSession, async_sessionmaker

from app.core.config import Settings, get_settings
from app.core.exceptions import InvalidCredentialsException, UserInactiveException
from app.core.security import decode_token
from app.crud.crud_user import get_user_by_id
from app.models.user import User

logger = logging.getLogger(__name__)
security = HTTPBearer(auto_error=True)


# ── SETTINGS DEPENDENCY ──────────────────────────────────
def get_settings_dep() -&gt; Settings:
    &quot;&quot;&quot;Inject Settings instance.

    This can be overridden in tests:
    app.dependency_overrides[get_settings_dep] = lambda: fake_settings
    &quot;&quot;&quot;
    return get_settings()


# ── DATABASE DEPENDENCY ──────────────────────────────────
async def get_db(request: Request) -&gt; AsyncGenerator[AsyncSession, None]:
    &quot;&quot;&quot;Inject AsyncSession from FastAPI app.state.session_maker.

    The session_maker must be set during app startup (in lifespan).

    This dependency:
    - Retrieves session_maker from app.state (set during startup)
    - Creates a new session per request
    - Automatically yields and closes the session
    - Can be overridden in tests via app.dependency_overrides[get_db]

    Args:
        request: FastAPI Request object (contains app.state)

    Yields:
        AsyncSession instance for database operations
    &quot;&quot;&quot;
    session_maker: async_sessionmaker[AsyncSession] = request.app.state.session_maker

    async with session_maker() as session:
        yield session


# ── AUTHENTICATION DEPENDENCY ────────────────────────────
async def get_current_user(
    credentials: Annotated[HTTPAuthorizationCredentials, Depends(security)],
    db: Annotated[AsyncSession, Depends(get_db)],
    settings: Annotated[Settings, Depends(get_settings_dep)],
) -&gt; User:
    &quot;&quot;&quot;Inject current authenticated user from JWT token.

    Args:
        credentials: JWT token from Authorization header
        db: Database session
        settings: Settings for JWT decoding

    Returns:
        Authenticated and active User instance

    Raises:
        InvalidCredentialsException: If token is invalid or user not found
        UserInactiveException: If user is not active
    &quot;&quot;&quot;
    payload = decode_token(credentials.credentials, settings.SECRET_KEY, settings.ALGORITHM)
    if payload is None:
        raise InvalidCredentialsException()

    user_id = payload.get(&quot;sub&quot;)
    if not isinstance(user_id, str):
        raise InvalidCredentialsException()

    user = await get_user_by_id(db, user_id)
    if user is None:
        logger.info(&quot;Token references missing user_id=%s&quot;, user_id)
        raise InvalidCredentialsException()

    if not user.is_active:
        raise UserInactiveException()

    return user

from datetime import timedelta
from typing import Annotated

from fastapi import APIRouter, Depends, status
from sqlalchemy.ext.asyncio import AsyncSession

from app.api.deps import get_db, get_settings_dep
from app.core.config import Settings
from app.core.exceptions import (
    EmailAlreadyRegistered,
    InvalidCredentialsException,
    UserInactiveException,
)
from app.core.security import create_access_token
from app.crud.crud_user import (
    authenticate_user,
    create_user,
    get_user_by_email,
)
from app.schemas.token import TokenResponse
from app.schemas.user import UserCreate, UserLogin, UserResponse

router = APIRouter()


@router.post(
    &quot;/register&quot;,
    response_model=UserResponse,
    status_code=status.HTTP_201_CREATED,
)
async def register(
    user_data: UserCreate,
    db: Annotated[AsyncSession, Depends(get_db)],
) -&gt; UserResponse:
    existing_user = await get_user_by_email(db, user_data.email)

    if existing_user is not None:
        raise EmailAlreadyRegistered()

    user = await create_user(
        db,
        email=user_data.email,
        password=user_data.password,
        full_name=user_data.full_name,
    )

    return UserResponse.model_validate(user)


@router.post(&quot;/login&quot;, response_model=TokenResponse)
async def login(
    credentials: UserLogin,
    db: Annotated[AsyncSession, Depends(get_db)],
    settings: Annotated[Settings, Depends(get_settings_dep)],
) -&gt; TokenResponse:
    user = await authenticate_user(
        db,
        credentials.email,
        credentials.password,
    )

    if user is None:
        raise InvalidCredentialsException()

    if not user.is_active:
        raise UserInactiveException()

    access_token = create_access_token(
        data={&quot;sub&quot;: str(user.id)},
        secret_key=settings.SECRET_KEY,
        algorithm=settings.ALGORITHM,
        expires_delta=timedelta(minutes=settings.ACCESS_TOKEN_EXPIRE_MINUTES),
    )

    return TokenResponse(access_token=access_token)

@router.post(
    &quot;/upload&quot;,
    response_model=DocumentResponse,
    status_code=status.HTTP_201_CREATED,

async def upload_document(
    file: Annotated[UploadFile, File()],
    current_user: Annotated[User, Depends(get_current_user)],
    db: Annotated[AsyncSession, Depends(get_db)],
    settings: Annotated[Settings, Depends(get_settings_dep)],
) -&gt; DocumentResponse:
    filename, content_type, file_content = await read_and_validate_upload(file)

    document = await create_document(
        db=db,
        user_id=current_user.id,
        filename=filename,
        content_type=content_type,
    )

    processing_document = await update_document_status(
        db=db,
        document_id=document.id,
        status=DocumentStatus.PROCESSING,
    )

    if processing_document is None:
        raise DocumentNotFound()

    document = processing_document

    # --- OBTENER HISTORIAL DE CORRECCIONES PARA CONTEXTO ---
    correction_history = await get_recent_corrections(
        db=db,
        user_id=current_user.id,
        limit=5,
    )

    try:
        # Pasamos el historial al orquestador
        extracted_data = await extract_document(
            file_bytes=file_content,
            filename=filename,
            content_type=content_type,
            settings=settings,
            correction_history=correction_history,  # &lt;-- NUEVO
        )

    except ValueError as exc:
        logger.info(
            &quot;Document validation failed for document_id=%s: %s&quot;,
            document.id,
            exc,
        )

        failed_document = await update_document_status(
            db=db,
            document_id=document.id,
            status=DocumentStatus.FAILED,
            error_logs=str(exc),
        )

        if failed_document is None:
            raise DocumentNotFound() from exc

        return DocumentResponse.model_validate(failed_document)

    except Exception as exc:
        logger.exception(
            &quot;Unexpected extraction error for document_id=%s&quot;,
            document.id,
        )

        failed_document = await update_document_status(
            db=db,
            document_id=document.id,
            status=DocumentStatus.FAILED,
            error_logs=&quot;Extraction failed unexpectedly&quot;,
        )

        if failed_document is None:
            raise DocumentNotFound() from exc

        return DocumentResponse.model_validate(failed_document)

    # ==========================================
    # NUEVA LÓGICA DE NEGOCIO SaaS (AUDITORÍA)
    # ==========================================
    score = extracted_data.confidence_score or 0.0

    # Decidimos el estado final basado en la confianza de nuestros adaptadores
    final_status = DocumentStatus.EXTRACTED
    if score &lt; 80.0:
        logger.info(&quot;Document %s routed to HUMAN REVIEW (Score: %s)&quot;, document.id, score)
        final_status = DocumentStatus.NEEDS_REVIEW

    extracted_document = await update_document_status(
        db=db,
        document_id=document.id,
        status=final_status,
        extracted_data=extracted_data.model_dump(),
        error_logs=None,
    )

    if extracted_document is None:
        raise DocumentNotFound()

    return DocumentResponse.model_validate(extracted_document)


@router.get(&quot;&quot;, response_model=list[DocumentResponse])
async def list_documents(
    skip: int = Query(default=0, ge=0),
    limit: int = Query(default=50, ge=1, le=100),
    patente: str | None = None,
    numero_guia: str | None = None,
    # ... más campos
    current_user: User = Depends(get_current_user),
    db: AsyncSession = Depends(get_db),
) -&gt; list[DocumentResponse]:
    # Construir la consulta dinámicamente
    query = select(Document).where(Document.user_id == current_user.id)
    
    if patente:
        # Filtro JSON: extracted_data-&gt;&gt;&apos;patente_camion&apos; = patente
        query = query.where(Document.extracted_data[&quot;patente_camion&quot;].astext == patente)
    if numero_guia:
        query = query.where(Document.extracted_data[&quot;numero_guia&quot;].astext == numero_guia)
    
    query = query.order_by(Document.created_at.desc()).offset(skip).limit(limit)
    result = await db.execute(query)
    documents = result.scalars().all()
    return [DocumentResponse.model_validate(doc) for doc in documents]

def build_engine(settings: Settings) -&gt; AsyncEngine:
    &quot;&quot;&quot;Create AsyncEngine instance based on settings.

import logging
import sys
from contextvars import ContextVar

request_id_context: ContextVar[str] = ContextVar(&quot;request_id&quot;, default=&quot;-&quot;)


class RequestIdFilter(logging.Filter):
    def filter(self, record: logging.LogRecord) -&gt; bool:
        record.request_id = request_id_context.get()
        return True


def configure_logging(level: str = &quot;INFO&quot;) -&gt; None:
    handler = logging.StreamHandler(sys.stdout)
    handler.addFilter(RequestIdFilter())
    handler.setFormatter(
        logging.Formatter(&quot;%(asctime)s %(levelname)s [%(request_id)s] %(name)s: %(message)s&quot;)
    )

    root_logger = logging.getLogger()
    root_logger.handlers.clear()
    root_logger.addHandler(handler)
    root_logger.setLevel(level.upper())

import time
import uuid
from collections import defaultdict, deque
from collections.abc import Callable

from fastapi import Request, Response, status
from starlette.middleware.base import BaseHTTPMiddleware

from app.core.config import settings
from app.core.logging import request_id_context


class RequestContextMiddleware(BaseHTTPMiddleware):
    async def dispatch(self, request: Request, call_next: Callable) -&gt; Response:
        request_id = request.headers.get(&quot;X-Request-ID&quot;, str(uuid.uuid4()))
        token = request_id_context.set(request_id)
        start = time.perf_counter()

        try:
            response = await call_next(request)
        finally:
            request_id_context.reset(token)

        response.headers[&quot;X-Request-ID&quot;] = request_id
        response.headers[&quot;X-Process-Time&quot;] = f&quot;{time.perf_counter() - start:.4f}&quot;
        return response


class InMemoryRateLimitMiddleware(BaseHTTPMiddleware):
    &quot;&quot;&quot;Small single-process limiter for MVP abuse protection.

    Use a reverse proxy or managed edge limiter once traffic runs across multiple
    API replicas.
    &quot;&quot;&quot;

    def __init__(self, app) -&gt; None:  # type: ignore[no-untyped-def]
        super().__init__(app)
        self.requests: defaultdict[str, deque[float]] = defaultdict(deque)

    async def dispatch(self, request: Request, call_next: Callable) -&gt; Response:
        if request.url.path in {&quot;/health&quot;, &quot;/ready&quot;}:
            return await call_next(request)

        now = time.monotonic()
        client_ip = request.client.host if request.client else &quot;unknown&quot;
        window = self.requests[client_ip]

        while window and now - window[0] &gt; settings.RATE_LIMIT_WINDOW_SECONDS:
            window.popleft()

        if len(window) &gt;= settings.RATE_LIMIT_REQUESTS:
            return Response(
                content=&apos;{&quot;detail&quot;:&quot;Rate limit exceeded&quot;}&apos;,
                media_type=&quot;application/json&quot;,
                status_code=status.HTTP_429_TOO_MANY_REQUESTS,
            )

        window.append(now)
        return await call_next(request)

# Use Argon2 instead of bcrypt to avoid passlib/bcrypt compatibility issues
pwd_context = CryptContext(schemes=[&quot;argon2&quot;], deprecated=&quot;auto&quot;)


def hash_password(password: str) -&gt; str:
    &quot;&quot;&quot;Hash password using argon2. Pure function, no dependencies.&quot;&quot;&quot;
    return pwd_context.hash(password)


def verify_password(plain_password: str, hashed_password: str) -&gt; bool:
    &quot;&quot;&quot;Verify password against bcrypt hash. Pure function, no dependencies.&quot;&quot;&quot;
    return pwd_context.verify(plain_password, hashed_password)


def create_access_token(
    data: dict[str, Any],
    secret_key: str,
    algorithm: str,
    expires_delta: timedelta | None = None,
) -&gt; str:
    &quot;&quot;&quot;Create JWT access token.

    Args:
        data: Payload data to encode
        secret_key: Secret key for signing
        algorithm: JWT algorithm (e.g., &quot;HS256&quot;)
        expires_delta: Optional expiration delta

    Returns:
        JWT token string
    &quot;&quot;&quot;
    expire = datetime.now(UTC) + (expires_delta or timedelta(minutes=30))
    payload = data.copy()
    payload.update({&quot;exp&quot;: expire})
    return jwt.encode(payload, secret_key, algorithm=algorithm)


def decode_token(token: str, secret_key: str, algorithm: str) -&gt; dict[str, Any] | None:
    &quot;&quot;&quot;Decode and verify JWT token.

        token: JWT token string
        secret_key: Secret key for verification
        algorithm: JWT algorithm (e.g., &quot;HS256&quot;)

    Returns:
        Decoded payload dict, or None if invalid
    &quot;&quot;&quot;
    try:
        return jwt.decode(token, secret_key, algorithms=[algorithm])
    except JWTError:
        return None

import logging
import re
from io import BytesIO
from typing import Any

from app.core.config import Settings
from app.schemas.extraction import ExtractedLogisticsData
from app.services.extractors.adapter_factory import AdapterFactory
from app.services.extractors.generic_llm_adapter import GenericLLMAdapter

# ----------------------------------------------
# LIMPIEZA DE TEXTO
# ----------------------------------------------
def clean_extracted_text(text: str) -&gt; str:
    &quot;&quot;&quot;
    Elimina líneas que parecen ruido: encabezados de tabla, líneas con solo números,
    líneas con mayúsculas sostenidas (ej. &quot;CONCEPTO&quot;, &quot;PORCENTAJE&quot;), etc.
    &quot;&quot;&quot;
    lines = text.split(&quot;\n&quot;)
    cleaned = []
    for line in lines:
        line = line.strip()
        if not line:
            continue
        # Saltar líneas que son solo números, guiones, barras, etc.
        if re.match(r&quot;^[\d\s\-\/\.\,\%\$]+$&quot;, line):
            continue
        # Saltar líneas en mayúsculas que parecen encabezados de tabla (más de 4 letras mayúsculas)
        if re.match(r&quot;^[A-Z\s]{5,}$&quot;, line):
            continue
        # Saltar líneas que empiezan con &quot;PAGO CON&quot;, &quot;CONCEPTO&quot;, etc.
        if re.match(r&quot;^(PAGO|CONCEPTO|PORCENTAJE|CÁLCULO|RETENCIONES)&quot;, line, re.IGNORECASE):
            continue
        # Saltar líneas que son etiquetas de tabla como &quot;Seguro Invalidez...&quot;
        if &quot;:&quot; not in line and len(line.split()) &gt; 5:
            continue
        cleaned.append(line)
    return &quot;\n&quot;.join(cleaned)


# ----------------------------------------------
# VALIDACIÓN SEMÁNTICA DE CAMPOS
# ----------------------------------------------
def validate_extracted_data(data: dict[str, Any]) -&gt; dict[str, Any]:
    &quot;&quot;&quot;
    Valida cada campo extraído y lo deja como None si no cumple con el formato esperado.
    &quot;&quot;&quot;

    def normalize_city(name: str) -&gt; str:
        return &quot; &quot;.join(word.capitalize() for word in name.lower().split())

    validated: dict[str, Any] = {}

    for campo, valor in data.items():
        if not valor or not isinstance(valor, str) or len(valor.strip()) &lt; 2:
            validated[campo] = None
            continue

        valor = valor.strip()

        if campo in (&quot;origen&quot;, &quot;destino&quot;):
            # Debe ser un nombre de ciudad (solo letras y espacios, no más de 3 palabras)
            if re.match(r&quot;^[A-ZÁÉÍÓÚÑ][a-záéíóúñ]+(?:\s+[A-ZÁÉÍÓÚÑ][a-záéíóúñ]+){0,2}$&quot;, valor):
                validated[campo] = normalize_city(valor)
            else:
                validated[campo] = None

        elif campo == &quot;chofer&quot;:
            if re.match(r&quot;^[A-ZÁÉÍÓÚÑ][a-záéíóúñ]+\s+[A-ZÁÉÍÓÚÑ][a-záéíóúñ]+$&quot;, valor):
                validated[campo] = valor
            else:
                validated[campo] = None

        elif campo == &quot;fecha_despacho&quot;:
            # Aceptar dd/mm/aaaa o dd-mm-aaaa
            if re.match(r&quot;^\d{2}[/-]\d{2}[/-]\d{4}$&quot;, valor):
                # Convertir a dd/mm/aaaa para consistencia
                if &quot;-&quot; in valor:
                    partes = valor.split(&quot;-&quot;)
                    valor = f&quot;{partes[0]}/{partes[1]}/{partes[2]}&quot;
                validated[campo] = valor
            else:
                validated[campo] = None

        elif campo == &quot;numero_guia&quot;:
            if re.match(r&quot;^\d+$&quot;, valor):
                validated[campo] = valor
            else:
                validated[campo] = None

        elif campo == &quot;patente_camion&quot;:
            if re.match(r&quot;^[A-Z]{2}-?[A-Z]{2}-?\d{2}$&quot;, valor) or re.match(
                r&quot;^[A-Z]{2}-?\d{4}$&quot;, valor
            ):
                validated[campo] = valor.upper()
            else:
                validated[campo] = None

        else:
            validated[campo] = valor

    return validated


# ----------------------------------------------
# EXTRACCIÓN DE TEXTO
# ----------------------------------------------
def extract_text_from_pdf(file_bytes: bytes) -&gt; str:
    try:
        import pdfplumber
    except ImportError:
        logger.warning(&quot;pdfplumber not installed&quot;)
        return &quot;&quot;

    try:
        text = []
        with pdfplumber.open(BytesIO(file_bytes)) as pdf:
            for page in pdf.pages:
                page_text = page.extract_text()
                if page_text:
                    text.append(page_text)
        return &quot;\n&quot;.join(text)
    except Exception:
        logger.exception(&quot;PDF extraction failed&quot;)
        return &quot;&quot;


def extract_text_from_image(file_bytes: bytes) -&gt; str:
    try:
        import pytesseract
        from PIL import Image
    except ImportError:
        logger.warning(&quot;OCR dependencies missing&quot;)
        return &quot;&quot;

    try:
        img = Image.open(BytesIO(file_bytes))
        return pytesseract.image_to_string(img, lang=&quot;spa&quot;)
    except Exception:
        logger.exception(&quot;Image OCR failed&quot;)
        return &quot;&quot;


def extract_text(file_bytes: bytes, content_type: str) -&gt; str:
    if content_type == &quot;application/pdf&quot;:
        return extract_text_from_pdf(file_bytes)
    if content_type in (&quot;image/png&quot;, &quot;image/jpeg&quot;, &quot;image/jpg&quot;):
        return extract_text_from_image(file_bytes)
    return &quot;&quot;

# ORQUESTADOR PRINCIPAL
# ----------------------------------------------
async def extract_document(
    file_bytes: bytes,
    filename: str,
    content_type: str,
    settings: Settings,
    correction_history: list[dict] | None = None,
) -&gt; ExtractedLogisticsData:

    if len(file_bytes) &lt; 20:
        raise ValueError(&quot;file too small&quot;)

    # 1. Extraer texto crudo
    text = extract_text(file_bytes, content_type)
    if not text:
        logger.info(&quot;No text extracted from %s&quot;, filename)
        return ExtractedLogisticsData(confidence_score=0.0, adapter_used=&quot;None&quot;)

    # 2. Limpiar el texto (eliminar ruido)
    cleaned_text = clean_extracted_text(text)
    logger.debug(&quot;Texto limpio (primeros 200 chars): %s&quot;, cleaned_text[:200])

    # 3. Obtener adaptador según clasificación
    adapter = AdapterFactory.get_adapter(cleaned_text)

    # 4. Extracción inicial
    raw_data = await adapter.extract_data(
        cleaned_text,
        image_bytes=file_bytes,
        settings=settings,
        correction_history=correction_history,
    )

    # 5. Validación semántica de los campos extraídos
    raw_data = validate_extracted_data(raw_data)

    # 6. Calcular confianza
    score = adapter.calculate_confidence(raw_data)

    # 7. Si confianza baja o faltan críticos, enriquecer con IA
    campos_criticos = [&quot;numero_guia&quot;, &quot;patente_camion&quot;]
    faltan_criticos = any(not raw_data.get(campo) for campo in campos_criticos)

    if (score &lt; 70 or faltan_criticos) and not isinstance(adapter, GenericLLMAdapter):
        logger.info(&quot;Confianza baja (%s) o faltan críticos. Enriqueciendo con IA genérica.&quot;, score)
        generic_adapter = GenericLLMAdapter()
        ai_data = await generic_adapter.extract_data(
            cleaned_text,
            image_bytes=file_bytes,
            settings=settings,
            correction_history=correction_history,
        )

        # Fusionar: los valores de la IA solo se usan si el adaptador específico no los tiene
        for key, value in ai_data.items():
            valor_actual = raw_data.get(key)
            if not valor_actual or (
                isinstance(valor_actual, str) and len(valor_actual.split()) &gt; 3
            ):
                if isinstance(value, str) and len(value.split()) &gt; 3:
                    continue
                raw_data[key] = value

        # Re-validar después de la fusión
        raw_data = validate_extracted_data(raw_data)

        raw_data[&quot;adapter_used&quot;] = f&quot;{adapter.__class__.__name__} + GenericLLM&quot;
        score = adapter.calculate_confidence(raw_data)
        raw_data[&quot;confidence_score&quot;] = score
        logger.info(&quot;Confianza después de enriquecimiento: %s&quot;, score)
    else:
        raw_data[&quot;confidence_score&quot;] = score
        if &quot;adapter_used&quot; not in raw_data:
            raw_data[&quot;adapter_used&quot;] = adapter.__class__.__name__

    # ==========================================
    # NUEVO: Extraer número de guía del nombre del archivo
    # ==========================================
    if not raw_data.get(&quot;numero_guia&quot;) and filename:
        # Buscar GDE-YYYY-NNNNN o GDE YYYY NNNNN en el nombre del archivo
        match = re.search(r&quot;GDE[- ]?(\d{4}[- ]?\d+)&quot;, filename, re.IGNORECASE)
        if match:
            raw_data[&quot;numero_guia&quot;] = match.group(1).replace(&quot; &quot;, &quot;&quot;).replace(&quot;-&quot;, &quot;&quot;)
            logger.info(f&quot;Número de guía extraído del archivo: {raw_data[&apos;numero_guia&apos;]}&quot;)

            # Recalcular confianza con el nuevo campo
            score = adapter.calculate_confidence(raw_data)
            raw_data[&quot;confidence_score&quot;] = score
            logger.info(f&quot;📊 Confianza recalculada: {score}&quot;)

    logger.info(
        &quot;Extracción finalizada. Adaptador: %s | Confianza: %s&quot;,
        raw_data.get(&quot;adapter_used&quot;),
        raw_data.get(&quot;confidence_score&quot;),
    )

    return ExtractedLogisticsData(**raw_data)

class DocumentClassifier:
    @staticmethod
    def identify_company(text: str) -&gt; str:
        text_upper = text.upper()

        # ---- DEPURACIÓN ----
        logger.debug(f&quot;🔍 Texto a clasificar (primeros 500 chars): {text[:500]}&quot;)

        # 1. DETECTAR GUÍA DE DESPACHO (PRIORIDAD ALTA)
        # Palabras clave
        keywords = [
            &quot;GDE&quot;,
            &quot;GUIA DE DESPACHO&quot;,
            &quot;GUÍA DE DESPACHO&quot;,
            &quot;FOLIO ELECTRONICO SII&quot;,
            &quot;FOLIO ELECTRÓNICO SII&quot;,
        ]
        for kw in keywords:
            if kw in text_upper:
                return &quot;chilean_guia&quot;

        # Patrón GDE-YYYY-NNNNN
        if re.search(r&quot;GDE\s*[-:]?\s*\d{4}\s*[-:]?\s*\d+&quot;, text_upper):
            return &quot;chilean_guia&quot;

        # Combinación de campos típicos
        has_patente = re.search(r&quot;PATENTE\s*VEH[IÍ]CULO&quot;, text_upper) is not None
        has_chofer = re.search(r&quot;CHOFER&quot;, text_upper) is not None
        has_fecha_emision = re.search(r&quot;FECHA\s*EMISI[ÓO]N&quot;, text_upper) is not None
        has_transportista = re.search(r&quot;TRANSPORTISTA&quot;, text_upper) is not None
        has_origen_destino = re.search(r&quot;DIRECCI[ÓO]N\s*DE\s*ORIGEN&quot;, text_upper) is not None

        if (
            sum([has_patente, has_chofer, has_fecha_emision, has_transportista, has_origen_destino])
            &gt;= 3
        ):
            return &quot;chilean_guia&quot;

        # 2. DETECTAR STARKEN (solo si NO es guía)
        if &quot;STARKEN&quot; in text_upper or &quot;TURBUS&quot; in text_upper:
            return &quot;starken&quot;

        # 3. Si no, devolver unknown
        return &quot;unknown&quot;


class AdapterFactory:
    @staticmethod
    def get_adapter(text: str) -&gt; BaseAdapter:
        company_id = DocumentClassifier.identify_company(text)

        adapters = {
            &quot;starken&quot;: StarkenAdapter(),
            &quot;chilean_guia&quot;: ChileanGuiaAdapter(),
        }

        if company_id in adapters:
            logger.info(
                f&quot;📌 Clasificado como &apos;{company_id}&apos; → &quot;
                f&quot;usando {adapters[company_id].__class__.__name__}&quot;
            )
            return adapters[company_id]

        logger.info(f&quot;❓ Documento no clasificado (tipo &apos;{company_id}&apos;) → usando GenericLLMAdapter&quot;)
        return GenericLLMAdapter()

from fastapi import HTTPException, UploadFile, status

from app.core.config import settings

PDF_MAGIC = b&quot;%PDF&quot;
JPEG_MAGIC = b&quot;\xff\xd8\xff&quot;
PNG_MAGIC = b&quot;\x89PNG\r\n\x1a\n&quot;


def sanitize_filename(filename: str | None) -&gt; str:
    if not filename:
        raise HTTPException(
            status_code=status.HTTP_400_BAD_REQUEST,
            detail=&quot;File name is required&quot;,
        )
    return Path(filename).name.replace(&quot;\x00&quot;, &quot;&quot;).strip()


def detect_content_type(file_bytes: bytes) -&gt; str:
    if file_bytes.startswith(PDF_MAGIC):
        return &quot;application/pdf&quot;
    if file_bytes.startswith(JPEG_MAGIC):
        return &quot;image/jpeg&quot;
    if file_bytes.startswith(PNG_MAGIC):
        return &quot;image/png&quot;
    raise HTTPException(
        status_code=status.HTTP_400_BAD_REQUEST,
        detail=&quot;Unsupported or invalid file content&quot;,
    )


async def read_and_validate_upload(file: UploadFile) -&gt; tuple[str, str, bytes]:
    filename = sanitize_filename(file.filename)
    extension = Path(filename).suffix.lower().lstrip(&quot;.&quot;)
    allowed_extensions = {item.lower() for item in settings.ALLOWED_EXTENSIONS}

    if extension not in allowed_extensions:
        raise HTTPException(
            status_code=status.HTTP_400_BAD_REQUEST,
            detail=f&quot;Unsupported file extension. Allowed: {&apos;, &apos;.join(sorted(allowed_extensions))}&quot;,
        )

    content = await file.read()
    if not content:
        raise HTTPException(status_code=status.HTTP_400_BAD_REQUEST, detail=&quot;File is empty&quot;)

    if len(content) &gt; settings.max_file_size_bytes:
        raise HTTPException(
            status_code=status.HTTP_413_REQUEST_ENTITY_TOO_LARGE,
            detail=f&quot;File exceeds {settings.MAX_FILE_SIZE_MB}MB limit&quot;,
        )

    content_type = detect_content_type(content)
    extension_content_map = {
        &quot;pdf&quot;: &quot;application/pdf&quot;,
        &quot;jpg&quot;: &quot;image/jpeg&quot;,
        &quot;jpeg&quot;: &quot;image/jpeg&quot;,
        &quot;png&quot;: &quot;image/png&quot;,
    }
    if extension_content_map.get(extension) != content_type:
        raise HTTPException(
            status_code=status.HTTP_400_BAD_REQUEST,
            detail=&quot;File extension does not match its content&quot;,
        )

    return filename, content_type, content
